"""
    date of creation : 04 Feb 2017.
    aim of script      : To understand the concept of for loops in python.
    coded by           : Rishikesh Agrawani.
"""
for i in range(1,11):   #Printing natural numbers from 1 to 10
    print i                    #Newline will be automatically appended 


for j in range(5,21):       #Printing numbers from 5 to 20
    print j


for k in range(2,21,2):     #Printing even numbers between 1 and 21, here 2 denotes the increment
    print k                         #in each iteration


for i in range(1,11):       #Printing tables for numbers in range [1,10]
    for j in range(1,11):   #Nested loop 
        print i*j,"\t",
    print "\n"

for c in "Rishikesh":       #Printing all characters of string 'Rishikesh' in a line separated
    print c,"\t",                 #by tab
print ""

for num in [1,3,7,9,11]:    #Printing all the numbers of list one by one in new lines
    print num
